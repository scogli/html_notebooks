Histograms

library(readxl)
## Warning: package 'readxl' was built under R version 3.6.2
library(ggplot2)
## Warning: package 'ggplot2' was built under R version 3.6.2
options(scipen = 9999)
 F <- read_excel("F.xlsx")
randomnormdata <- rnorm(1270)

hist(randomnormdata,main = "Histogram of Random Normal Distribution",
     xlab = "Random Normal Distribution",
     col = "darkseagreen2",
     border = "grey")

rt <- diff(log(F$`Adj Close`))
hist(rt,main = "Histogram of Ford Stock Returns",
     xlab = "Stock Returns",
     col = "goldenrod",
     border = "white")

Scatter Plots

ggplot()+
  geom_point(aes(x = F$Date[2:1271],y = rt),colour = "dodgerblue",alpha = 0.55)+
  ggtitle("Scatter Plot of Ford's Stock Returns")+
  labs(x="Date",y="Logarithmic Returns",
       size = "Volume",
       subtitle = "Period 2010-2020")

ggplot()+
  geom_point(aes(x = F$Date[2:1271],y = rt, size = F$Volume[2:1271]),colour = "dodgerblue",alpha = 0.55)+
  ggtitle("Scatter Plot of Ford's Stock Returns")+
  labs(x="Date",y="Logarithmic Returns",
